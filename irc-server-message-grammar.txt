IRC Server Message Grammar

<message> ::= <server-message> | <client-message>

<server-message> ::= 
	<source> <numeric-reply> | 
	<source> <notice> | 
	"PING" <payload>

<numeric-reply> ::= <response-code> <nick> <message>

<client-message> ::= <name> "PRIVMSG" <channel> <payload>

<source> ::= ":" <name>

<channel> ::= REGEX("[&#!+][a-zA-Z0-9.]*")

<payload> ::= ":" <text>

<name> ::= REGEX("[a-zA-Z0-9!@.]*")

<notice> ::= <source> "NOTICE" <payload>

<text> ::= it's text motherfucker
